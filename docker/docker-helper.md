## Docker полезные команды:
```bash
# Залогиниться на сервере
docker login <IP>:<PORT>
# Запуллить удаленный образ docker по $URL контейнера в виде <репозиторий>:<название>
docker pull <репозиторий>:<название>
# Запустить в интерактивном режиме контейнер
docker run -it <репозиторий>:<название> /bin/bash
# Когда поменяли что-то жмём Ctrl+D и надо закоммитить
docker ps -a
# Там будет сводная информация по контейнерам. Нужен первый столбец CONTAINER ID
# Коммитим
docker commit -m "Описание" -a "Автор" id_коммита <репозиторий>:<название>
# Репозиторий это URL, где лежал контейнер
# Пушим (заливаем) контейнер
docker push <репозиторий>
# Удалить контейнер
sudo docker rmi -f <репозиторий>:<название> #можно через id контейнера
# Как изменить контейнер
sudo pull <репозиторий>:<название>
sudo docker run -it <репозиторий>:<название> /bin/bash
# меняем что-то
sudo docker ps -a
sudo docker commit <container_id> <репозиторий>:<название> # можно sudo docker commit -m "text" -a "author commit" <container_id> <репозиторий>:<название>
sudo docker push <репозиторий> #может можно и <репозиторий>:<название>
# Скопировать что-то на запущенный контейнер
sudo docker cp ./Local_file <container_id>:/path
# Скопировать что-то из запущенного контейнера
sudo docker cp <container_id>:/path ./local_path
```
